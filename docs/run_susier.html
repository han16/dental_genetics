  cat("SNP:", locus_id, "is running", "\n")

  # Parse Chr and Pos from filename
  base <- sub("\\.ld$", "", locus_id) 
  chr <- as.integer(sub("_.*", "", base)) 
  pos <- as.integer(sub(".*_", "", base))

  # Read the LD square matrix
  # Note: adjust 'header' and 'check.names' based on the  file format
  snplist_file <- sub("\\.ld$", ".snplist", ld_file)
  snp_names <- readLines(snplist_file)
  
  # Keep only the first name if multiple are present (separated by ;)
  snp_names <- sub(";.*", "", snp_names)
  
  R_raw <- as.matrix(read.table(ld_file, header = FALSE))
  sum(is.na(R_raw))
  rownames(R_raw) <- colnames(R_raw) <- snp_names
  
  ######### map_file is to map snp id, rsXXXX to coordinate chr:pos since gwas uses coordinate 
  map_file     <- sub("\\.ld$", ".id_map", ld_file)
  #  Read the ID Map (rsID to chr:pos)
  id_lookup <- read.table(map_file, header = FALSE, stringsAsFactors = FALSE)
  colnames(id_lookup) <- c("rsid", "coords")
  
  # Translate rsIDs to chr:pos based on the map, some are not matched, leading to NA's 
  matched_indices <- match(snp_names, id_lookup$rsid)
  chr_pos_names   <- id_lookup$coords[matched_indices]
  
  # Assign the new chr:pos names to LD matrix
  rownames(R_raw) <- colnames(R_raw) <- chr_pos_names
  snps_in_ld <- colnames(R_raw)
  
  ############################
  # Align GWAS and LD
  ############################
  gwas_locus <- gwas[
      CHR == chr & 
      BP >= (pos - WINDOW) & 
      BP <= (pos + WINDOW)
    ]

  if (nrow(gwas_locus) < 10) {
    warning("Too few GWAS SNPs, skipping ", locus_id)
    next
  }

  # Find overlap between GWAS MarkerName and LD matrix names
  common_snps <- intersect(gwas_locus$MarkerName, snps_in_ld)
  common_snps[1:10]

  if (length(common_snps) < 10) {
    warning("Too few overlapping SNPs, skipping ", locus_id)
    next
  }

  # Subset and Reorder both to match exactly
  # 1. Subset GWAS and set key for fast indexing
  gwas_locus <- gwas_locus[MarkerName %in% common_snps]
  setkey(gwas_locus, MarkerName)
  
  # 2. Subset and Reorder Matrix
  R <- R_raw[common_snps, common_snps]
  na_snps <- colSums(is.na(R)) > 0
# Remove them
  R <- R[!na_snps, !na_snps]


  # 3. Final alignment: Ensure GWAS rows are in the exact same order as R columns
  gwas_locus <- gwas_locus[colnames(R)]

  # Safety check
  stopifnot(identical(gwas_locus$MarkerName, colnames(R)))

  ############### run susieR ###########
  z <- gwas_locus$Effect / gwas_locus$StdErr 

  fit <- susie_rss(
    z = z,
    R = R,
    n = 26533,
    L = L_MAX, 
    estimate_prior_method = "EM",
    estimate_residual_variance = FALSE
  )

  susie_results[[i]] <- list(
      locus = paste0("chr", chr, ":", pos),
      fit   = fit,
      gwas  = gwas_locus
  )
}

# 4. Define a helper function to find the nearest gene
find_nearest_gene <- function(chr, pos, genes_df) {
  # Filter genes on the same chromosome
  chrom_genes <- genes_df[genes_df$chromosome_name == as.character(chr), ]
  
  if (nrow(chrom_genes) == 0) return(c(NA, NA))
  
  # Calculate distance to gene start/end (whichever is closer)
  dist_start <- abs(chrom_genes$start_position - pos)
  dist_end   <- abs(chrom_genes$end_position - pos)
  min_dist   <- pmin(dist_start, dist_end)
  
  nearest_idx <- which.min(min_dist)
  return(c(chrom_genes$ensembl_gene_id[nearest_idx], 
           chrom_genes$external_gene_name[nearest_idx]))
}

fetch_eqtl <- function(rsid, gene_id, tissue) {
